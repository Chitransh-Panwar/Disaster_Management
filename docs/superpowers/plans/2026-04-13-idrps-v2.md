# IDRPS v2 (Pan-India Rescue Planner) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-based Leaflet + OpenStreetMap app with sliding tool/data panels, 12 marker types, 5 preloaded scenarios, and visualized CP algorithms (Dijkstra, Tarjan bridges, DSU connectivity, BFS spread, Knapsack, LCA).

**Architecture:** Vanilla JS app split into (1) UI modules (panels, forms, log), (2) map renderer (Leaflet layers + interactions), (3) domain state/store (markers/resources/roads), and (4) pure algorithm modules (testable in Node).

**Tech Stack:** HTML5, CSS3, Vanilla JS (ES modules), Leaflet.js v1.9, OpenStreetMap tiles, `localStorage`; optional Node built-in test runner (`node --test`) for algorithm tests; optional `python3 -m http.server` for local serving.

```js
export function initMap(mapElId = 'map') {
  if (typeof L === 'undefined') {

}
```

```js
export function initPanels() {
  const leftPanel = document.getElementById('leftPanel');
  const rightPanel = document.getElementById('rightPanel');
  const leftToggle = document.getElementById('leftToggle');
  const rightToggle = document.getElementById('rightToggle');

  let leftOpen = false;
  let rightOpen = false;

  function syncPanels() {
    leftPanel.classList.toggle('open', leftOpen);
    rightPanel.classList.toggle('open', rightOpen);

    leftToggle.setAttribute('aria-expanded', String(leftOpen));
    rightToggle.setAttribute('aria-expanded', String(rightOpen));

    leftToggle.textContent = leftOpen ? 'Close ▶' : '◀ Tools';
    rightToggle.textContent = rightOpen ? '◀ Close' : '▶ Data';
  }

  leftToggle.addEventListener('click', () => {
    leftOpen = !leftOpen;
    syncPanels();
  });

  rightToggle.addEventListener('click', () => {
    rightOpen = !rightOpen;
    syncPanels();
  });

  // Tabs (use existing ARIA markup: role="tab" + aria-controls)
  const tabButtons = Array.from(rightPanel.querySelectorAll('[role="tab"]'));

  function setActiveTab(activeBtn) {
    for (const btn of tabButtons) {
      const isActive = btn === activeBtn;
      btn.setAttribute('aria-selected', String(isActive));

      const panelId = btn.getAttribute('aria-controls');
      const panelEl = panelId ? document.getElementById(panelId) : null;
      if (panelEl) panelEl.hidden = !isActive;
    }
  }

  tabButtons.forEach((btn) => btn.addEventListener('click', () => setActiveTab(btn)));

  syncPanels();
  setActiveTab(tabButtons.find((b) => b.getAttribute('aria-selected') === 'true') ?? tabButtons[0]);
}
```

```js
export function createEventLog(targetEl) {
  const entries = [];

  function ts() {
    return new Date().toISOString().slice(11, 19); // HH:MM:SS
  }

  function logEvent(type, message, meta = {}) {
    const entry = { time: ts(), type, message, meta };
    entries.unshift(entry);

    const row = document.createElement('div');
    row.textContent = `[${entry.time}] ${entry.type}: ${entry.message}`;
    targetEl.prepend(row);
  }

  return { logEvent, entries };
}
```

```js
export function dijkstra(graph, start, goal) {
  const dist = new Map();
  const prev = new Map();
  const visited = new Set();

  for (const k of Object.keys(graph)) dist.set(k, Infinity);
  dist.set(start, 0);

  while (true) {
    let u = null;
    let best = Infinity;
    for (const [node, d] of dist.entries()) {
      if (!visited.has(node) && d < best) {
        best = d;
        u = node;
      }
    }
    if (u === null) break;
    if (u === goal) break;

    visited.add(u);
    for (const e of graph[u] ?? []) {
      const nd = dist.get(u) + e.w;
      if (nd < dist.get(e.to)) {
        dist.set(e.to, nd);
        prev.set(e.to, u);
      }
    }
  }

  const distance = dist.get(goal);
  if (!Number.isFinite(distance)) return { distance: Infinity, path: [] };

  const path = [];
  for (let cur = goal; cur != null; cur = prev.get(cur)) {
    path.push(cur);
    if (cur === start) break;
  }
  path.reverse();
  return { distance, path };
}
```

```js
export async function loadRoadNetwork(url = 'data/india_roads.json') {
  const res = await fetch(url);
  if (!res.ok) throw new Error(`Failed to load road network: ${res.status}`);
  const json = await res.json();
  return json;
}

export function buildAdjacency(network) {
  const adj = {};
  for (const n of network.nodes) adj[n.id] = [];

  for (const e of network.edges) {
    if (e.status === 'blocked') continue;
    const w = e.km * (e.status === 'partial' ? 5 : 1);
    adj[e.from].push({ to: e.to, w, edgeId: e.id });
    adj[e.to].push({ to: e.from, w, edgeId: e.id });
  }

  return adj;
}
```

```js
export function createRoadLayer(map) {
  const group = L.layerGroup().addTo(map);
  return {
    group,
    clear() { group.clearLayers(); },
    render(network) {
      group.clearLayers();
      const byId = new Map(network.nodes.map(n => [n.id, n]));

      for (const e of network.edges) {
        const a = byId.get(e.from);
        const b = byId.get(e.to);
        if (!a || !b) continue;
```

```js
export function createStore(reducer, preloadedState) {
  let state = preloadedState;
  /** @type {Array<() => void>} */
  const listeners = [];

  function getState() {
    return state;
  }

  function dispatch(action) {
    state = reducer(state, action);
    for (const l of listeners) l();
    return action;
  }

  function subscribe(listener) {
    listeners.push(listener);
    return () => {
      const idx = listeners.indexOf(listener);
      if (idx >= 0) listeners.splice(idx, 1);
    };
  }

  dispatch({ type: '@@INIT' });
  return { getState, dispatch, subscribe };
}
```

```js
    route.clear();
    return;
  }

  // Snap to nearest road node: for MVP, use the closest node by straight-line distance
  const byId = new Map(net.nodes.map(n => [n.id, n]));
  const dist2 = (aLat, aLng, bLat, bLng) => {
    const dx = aLat - bLat; const dy = aLng - bLng; return dx*dx + dy*dy;

      if (d < bestD) { bestD = d; best = n.id; }
    }
    return best;

  const gId = nearestNodeId(goal.lat, goal.lng);

  const res = dijkstra(adj, sId, gId);
  eventLog.logEvent('dijkstra', `Distance: ${Math.round(res.distance)} km, path nodes: ${res.path.length}`);
```

```js
export class DSU {
  constructor(items) {
    this.parent = new Map();
    this.rank = new Map();
    for (const x of items) {
      this.parent.set(x, x);
      this.rank.set(x, 0);
    }
  }

  find(x) {
    const p = this.parent.get(x);
    if (p === x) return x;
    const r = this.find(p);
    this.parent.set(x, r);
    return r;
  }

  union(a, b) {
    let ra = this.find(a);
    let rb = this.find(b);
    if (ra === rb) return false;

    const rka = this.rank.get(ra);
    const rkb = this.rank.get(rb);

    if (rka < rkb) [ra, rb] = [rb, ra];
    this.parent.set(rb, ra);
    if (rka === rkb) this.rank.set(ra, rka + 1);
    return true;
  }

  components() {
    const roots = new Set();
    for (const k of this.parent.keys()) roots.add(this.find(k));
    return roots.size;
  }
}
```

- [ ] **Step 2: Implement `js/algo/tarjanBridges.js`**

```js
export function findBridgeEdgeIds(adj) {
  const disc = new Map();
  const low = new Map();
  let time = 0;
  const bridges = new Set();

  function dfs(u, parent) {
    time += 1;
    disc.set(u, time);
    low.set(u, time);

    for (const e of (adj[u] ?? [])) {
      const v = e.to;
      if (!disc.has(v)) {
        dfs(v, u);
        low.set(u, Math.min(low.get(u), low.get(v)));
        if (low.get(v) > disc.get(u)) bridges.add(e.edgeId);
      } else if (v !== parent) {
        low.set(u, Math.min(low.get(u), disc.get(v)));
      }
    }
  }

  for (const u of Object.keys(adj)) {
    if (!disc.has(u)) dfs(u, null);
  }

  return Array.from(bridges);
}
```

```js
export function bfsLevels(adj, start) {
  const q = [start];
  const levels = new Map([[start, 0]]);

  while (q.length) {
    const u = q.shift();
    const lu = levels.get(u);
    for (const e of (adj[u] ?? [])) {
      const v = e.to;
      if (levels.has(v)) continue;
      levels.set(v, lu + 1);
      q.push(v);
    }
  }

  return levels;
}
```
